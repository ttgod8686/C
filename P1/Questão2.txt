#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main()
{
    float i, n, aux = -1, s, soma;
    
    printf("Oual valor de N? ");
    scanf ("%f",&n);
    
    for(i = 0; i == n ; i++){
    s = (i * n) /  (n - i) * (n - i);
    soma = soma + s * aux;
    aux = aux * -1;
    }
    printf("\n%.0f", soma);

    return 0;
}
